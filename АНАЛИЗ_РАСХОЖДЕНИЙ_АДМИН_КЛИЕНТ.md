## 3. Система доступа и покупок

### 🟢 В админке

#### Множественные тарифные планы

Каждый курс может иметь несколько тарифов:

```typescript
// Пример из courseData.ts
pricingPlans: [
  {
    id: 'plan-1month',
    courseId: 'brow-master-pro',
    name: '1 месяц',
    price: 18000,
    accessPeriod: { type: '1month', days: 30 }
  },
  {
    id: 'plan-2months',
    courseId: 'brow-master-pro',
    name: '2 месяца',
    price: 27000,
    isRecommended: true // 🌟 Рекомендуемый
  },
  {
    id: 'plan-unlimited',
    courseId: 'brow-master-pro',
    name: 'Бессрочный',
    price: 45000,
    accessPeriod: { type: 'unlimited', days: null }
  }
]
```

**Логика доступа в админке:**
- Каждый курс независим
- Каждый тарифный план привязан к конкретному курсу
- Можно создавать неограниченное количество планов
- Админ контролирует, какие планы активны

### 🟡 В клиентской части

#### [`hooks/usePurchase.ts`](hooks/usePurchase.ts:23)

```typescript
export const usePurchase = (courseId: string = 'brow-master-pro')
```

**Проблемы:**

1. **По умолчанию всегда `brow-master-pro`**
   ```typescript
   // В большинстве мест вызывается без параметра
   const { isPurchased, hasAccess } = usePurchase();
   // ❌ Означает: всегда проверяем только brow-master-pro
   ```

2. **Функция `hasAccess` принимает courseId, но везде передается некорректно**
   ```typescript
   // В ModuleList.tsx (строка 31)
   const accessible = hasAccess(lesson.isFree);
   // ❌ ОШИБКА: передается boolean вместо courseId
   
   // Правильно должно быть:
   const accessible = hasAccess(courseId, lesson.isFree);
   ```

3. **Жесткая привязка в компонентах**
   ```typescript
   // components/dashboard/ModuleList.tsx
   import { courseData } from '@/lib/courseData';
   // ❌ Всегда импортируется ОДИН курс
   
   // Правильно:
   // Должен передаваться courseId через props
   ```

## 5. Навигация и пользовательский опыт

### 🔀 Навигация в админке

```
/admin
├── /courses                    # Список ВСЕХ курсов
│   ├── /create                 # Создать новый курс
│   └── /[courseId]             # Конкретный курс
│       ├── /edit               # Редактировать курс
│       └── /pricing            # Управление тарифами
│           ├── /create         # Создать тариф
│           └── /[planId]/edit  # Редактировать тариф

│   ├── /create                 # Создать модуль (с выбором курса)

    ├── /create                 # Создать урок (с выбором модуля)
```

**Особенности:**
- ✅ Иерархическая структура
- ✅ Все операции контекстно зависимы от курса
- ✅ Создавая модуль, админ выбирает курс
- ✅ Создавая урок, админ выбирает модуль (который привязан к курсу)


```tsx
// СТРОКА 5 - Жесткий импорт
import { courseData } from '@/lib/courseData';
```

**Исправленная версия:**
```tsx
interface ModuleListProps {
  courseId: string; // ✅ Принимаем courseId через props
}

export default function ModuleList({ courseId }: ModuleListProps) {
  const course = getCourseById(courseId);
  const { hasAccess } = usePurchase(courseId); // ✅ Передаем courseId
  
  if (!course) return <div>Курс не найден</div>;
  
  return (
    <div>
      {course.modules.map((module) => (
        <div key={module.id}>
          {module.lessons.map((lesson) => {
            const accessible = hasAccess(courseId, lesson.isFree);
            // ✅ Правильная проверка доступа
          })}
        </div>
      ))}
    </div>
  );
}
```

### Пример 2: Обновленный `hooks/useProgress.ts`

```typescript
'use client';

import { useState, useEffect } from 'react';
import { getAllLessons } from '@/lib/courseData';

interface ProgressByCourse {
  [courseId: string]: string[];
}

export const useProgress = (courseId?: string) => {
  const [progress, setProgress] = useState<ProgressByCourse>({});

  useEffect(() => {
    // Загрузка прогресса из localStorage
    const loadProgress = () => {
      if (typeof window === 'undefined') return;
      
      const stored = localStorage.getItem('nail_progress');
      if (stored) {
        try {
          const parsed = JSON.parse(stored);
          
          // Миграция старого формата
          if (Array.isArray(parsed)) {
            const migrated = { 'brow-master-pro': parsed };
            localStorage.setItem('nail_progress', JSON.stringify(migrated));
            setProgress(migrated);
          } else {
            setProgress(parsed);
          }
        } catch (error) {
          console.error('Ошибка загрузки прогресса:', error);
        }
      }
    };
    
    loadProgress();
  }, []);

  const saveProgress = (newProgress: ProgressByCourse) => {
    if (typeof window === 'undefined') return;
    localStorage.setItem('nail_progress', JSON.stringify(newProgress));
    setProgress(newProgress);
  };

  const isLessonCompleted = (lessonId: string, cId?: string): boolean => {
    const targetCourseId = cId || courseId || 'brow-master-pro';
    return progress[targetCourseId]?.includes(lessonId) || false;
  };

  const markLessonCompleted = (lessonId: string, cId?: string): void => {
    const targetCourseId = cId || courseId || 'brow-master-pro';
    const courseProgress = progress[targetCourseId] || [];
    
    if (!courseProgress.includes(lessonId)) {
      const newProgress = {
        ...progress,
        [targetCourseId]: [...courseProgress, lessonId]
      };
      saveProgress(newProgress);
    }
  };

  const getProgressPercentage = (cId?: string): number => {
    const targetCourseId = cId || courseId;
    
    if (!targetCourseId) {
      // Общий прогресс по всем курсам
      const allLessons = getAllLessons();
      const allCompleted = Object.values(progress).flat();
      return Math.round((allCompleted.length / allLessons.length) * 100);
    }
    
    const courseLessons = getAllLessons(targetCourseId);
    const courseProgress = progress[targetCourseId] || [];
    return Math.round((courseProgress.length / courseLessons.length) * 100);
  };

  const getCompletedCount = (cId?: string): number => {
    const targetCourseId = cId || courseId;
    
    if (!targetCourseId) {
      return Object.values(progress).flat().length;
    }
    
    return progress[targetCourseId]?.length || 0;
  };

  const getTotalCount = (cId?: string): number => {
    const targetCourseId = cId || courseId;
    const lessons = getAllLessons(targetCourseId);
    return lessons.length;
  };

  const isFullyCompleted = (cId?: string): boolean => {
    return getProgressPercentage(cId) === 100;
  };

  return {
    isLessonCompleted,
    markLessonCompleted,
    getProgressPercentage,
    getCompletedCount,
    getTotalCount,
    isFullyCompleted,
  };
};
```
